1function T = Control_Block(omega, omega_des, q, q_des)
 2%This is a function that emulates the Controller Block of the simulink
 3%pointing controller.
 4
 5kw=1;
 6kq=1;
 7
 8omega_gain=(-kw)*(omega-omega_des);
 9
10q1_des=q_des(1);
11q2_des=q_des(2);
12q3_des=q_des(3);
13q4_des=q_des(4);
14
15Xi_qtilde=[q4_des  -q3_des   q2_des;
16           q3_des   q4_des  -q1_des;
17          -q2_des   q1_des   q4_des;
18          -q1_des  -q2_des  -q3_des]; 
19      
20Xi_qtilde_trans=transpose(Xi_qtilde);
21
22q_prod=Xi_qtilde_trans*q;
23q_gain=(-kq)*q_prod;
24
25T=omega_gain+q_gain;
26
27end
