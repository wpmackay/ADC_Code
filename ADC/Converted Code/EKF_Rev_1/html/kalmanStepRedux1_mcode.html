1function [q_hat_out, w_hat, theta_hat, psi_hat, phi_hat, sig3_out, B_hat_out, P_out] = kalmanStepRedux(B_in, B_meas_in, rr_meas_in, q_meas_in, trackerAvail, w_meas_in, q_hat_in, P_in, B_hat_in)
 2
 3%% Performs one kalman operation.
 4%% Inputs:
 5%   B_in - 1x3 - IGRF components
 6%   B_meas_in - 1 x 3 - Magnetometer measurements
 7%   rr_meas -   1 x 3 - star tracker reference vector measurement
 8%   q_hat_in -  4 x 1 - Previous estimated quaternion
 9%   q_meas_in - 4 x 1 - star tracker measured quaternion
10%   P_in -      6 x 6 - Previous covariance matrix
11%   w_meas_in - 3 x 1 - Gyro rate measurement in rad/s
12%   B_hat_in  - 3 x 1 - Previous Gyro bias estimate
13%   trackerAvail- 1 x 1 - 0 or 1 value indicating whether a star tracker
14%   measurement is available
15%% Outputs:
16%   q_hat_out - 4 x 1 - Current Estimated Quaternion
17%   B_hat_out - 3 x 1 - Current Gyro Bias Estimate
18%   sig3_out  - 6 x 1 - 3 sigma bounds for Euler Angle error
19%   theta_hat - 1 x 1 - Theta angle in degrees
20%   psi_hat   - 1 x 1 - Psi angle in degrees
21%   phi_hat   - 1 x 1 - Phi angle in degrees
22%   w_hat     - 3 x 1 - estimated angular rates in rad/s
23%   P_out     - 6 x 6 - Current covarian
24
25sig_b = 400;
26sig_st = 1.1010e-04;
27sig_v = 6.0711e-05;
28sig_u = 6.0711e-08;
29dt = 0.1;
30
31Q=[sig_v^2*eye(3), zeros(3); zeros(3), sig_u^2*eye(3)];
32
33if trackerAvail == 1 %&& t(k)~=0 %If a star tracker measurement is available
34    R=[100*sig_b*eye(3)',zeros(3);zeros(3), sig_st*eye(3)'];
35else
36    R=[100*sig_b*eye(3)'];
37end
38% If weights of the star tracker and magnetometers are adjusted, can tune
39% filter performance.
40
41% Gain
42A_hat=q2a(q_hat_in);
43if trackerAvail == 1 %&& t(k)~=0 %If a star tracker measurement is available
44    H=[cpm(A_hat*B_in'),zeros(3);
45        cpm(A_hat*rr_meas_in'),zeros(3)];
46else
47    H=[cpm(A_hat*B_in'),zeros(3)];
48end
49
50K=P_in*H'*(H*P_in*H'+R)^-1;
51
52% Update
53P_in=(eye(6)-K*H)*P_in;
54if trackerAvail == 1 %&& t(k)~=0 %If a star tracker measurement is available
55    h=[A_hat*B_in';
56        A_hat*rr_meas_in'];
57    y=[B_meas_in';
58        q2a(q_meas_in)*rr_meas_in'];%Measurement
59else
60    h=[A_hat*B_in'];
61    y=[B_meas_in'];%Measurement
62end
63
64dx=K*(y-h);
65da=dx(1:3);
66db=dx(4:6);
67q_hat_in=q_hat_in+0.5*Xi(q_hat_in)*da;
68q_hat_in=q_hat_in/norm(q_hat_in); %Re-normalize q
69B_hat_in=B_hat_in+db;
70
71% Propagation
72w_hat=w_meas_in-B_hat_in;
73F=[-cpm(w_hat), -eye(3); zeros(3,3), zeros(3,3)];
74G=[-eye(3) zeros(3,3); zeros(3,3), eye(3)];
75P_dot=F*P_in+P_in*F'+G*Q*G';
76
77% Next time step:
78%     q_hat_out=q_hat_in+dt*q_dot;
79PSI=sin(0.5*norm(w_hat)*dt)*w_hat/norm(w_hat);
80OMEGA=[cos(0.5*norm(w_hat)*dt)*eye(3)-cpm(PSI), PSI; -PSI', cos(0.5*norm(w_hat)*dt)]; %cpm returns cross product matrix
81q_hat_out=OMEGA*q_hat_in;
82
83P_out=P_in+dt*P_dot;
84B_hat_out = B_hat_in;
85
86%3-sigma bounds
87sig3_out=rad2deg(diag(P_out).^0.5*3);
88
89%Estimated Euler angles for each time step:
90theta_hat=acosd(A_hat(3,3)); %degrees
91psi_hat=acosd(A_hat(3,2)/sind(theta_hat)); %degrees
92phi_hat=acosd(A_hat(2,3)/-sind(theta_hat)); %degrees
