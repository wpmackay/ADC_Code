1% y = averaging_filter(x)
 2% Take an input vector signal 'x' and produce an output vector signal 'y' with
 3% same type and shape as 'x' but filtered.
 4function y = averaging_filter(x) %#codegen
 5% Use a persistent variable 'buffer' that represents a sliding window of
 6% 16 samples at a time.
 7persistent buffer;
 8if isempty(buffer)
 9    buffer = zeros(16,1);
10end
11y = zeros(size(x), class(x));
12for i = 1:numel(x)
13    % Scroll the buffer
14    buffer(2:end) = buffer(1:end-1);
15    % Add a new sample value to the buffer
16    buffer(1) = x(i);
17    % Compute the current average value of the window and
18    % write result
19    y(i) = sum(buffer)/numel(buffer);
20end
